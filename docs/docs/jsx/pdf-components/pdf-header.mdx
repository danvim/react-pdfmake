## Example

import { PdfHeader } from "react-pdfmake-reconciler";

const staticHeader = (
  <PdfHeader>

  <PdfHeader>
    {(currentPage, pageCount, pageSize) => (
      <pdf-text>
        Page {currentPage} / {pageCount}
      </pdf-text>
    )}
  </PdfHeader>
);
