## Example

```tsx twoslash
import { PdfFooter } from "react-pdfmake-reconciler";

const staticFooter = (
  <PdfFooter>
    <pdf-text>Hello world!</pdf-text>
  </PdfFooter>
);

const dynamicFooter = (
  <PdfFooter>
    {(currentPage, pageCount, pageSize) => (
      <pdf-text>
        Page {currentPage} / {pageCount}
      </pdf-text>
    )}
  </PdfFooter>
);
```
